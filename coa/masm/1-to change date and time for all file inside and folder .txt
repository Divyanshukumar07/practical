Get-ChildItem | ForEach-Object {
    $_.CreationTime  = "7 January 2026 11:09:51" 
    $_.LastWriteTime = "7 January 2026 11:09:51" 
    $_.LastAccessTime= "7 January 2026 11:09:51" 
}
