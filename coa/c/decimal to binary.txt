#include <stdio.h>

int main()
{
    int n, i;

    printf("Enter a number: ");
    scanf("%d", &n);

    printf("Binary equivalent: ");
    for (i = 31; i >=0; i--) //i=31 means 32 bit if i put i = 3 then it means it will display 4 bit
    {
        printf("%d", (n >> i) & 1);
    }

    return 0;
}
